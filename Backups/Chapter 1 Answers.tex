				\item general solution
					\begin{align}
						\dfrac{1}{p-900} dp = \dfrac{1}{2}dt\\
						\diff{}{t} \left(\ln \left| p-900 \right| \right) = dt \\
						\intertext{Integrate both sides to obtain} 
						\ln \left| p-900 \right| = t + C \\
						e^{t + C} = p-900
						\intertext{$e^{t+C}$ gives $ke^t$ }
						p = ke^{t/2} + 900
 					\end{align}
